float s,a,area;

scanf("%d%d%d",&s1,&s2,&s3);
if((s1+s2)>s3&&(s2+s3)>s1&&(s3+s1)>s2)
{
if(s1==s2&&s2==s3)
printf("Equilateral Triangle\n");
else if(s1==s2||s2==s3||s3==s1)
printf("Isosceles Triangle\n");
else printf("Scalene Triangle\n");

s=(s1+s2+s3)/2.0;
printf("Semiperimeter :%f\n",s);
a=s*(s-s1)*(s-s2)*(s-s3);
area=pow(a,0.5);
printf("Area of Triangle : %f\n",area);
}
else
printf("Triangle does not exist !!!");
getch();
}

void main()
{
int n,r,factn=1,factr=1,factnmr=1,i,npr,ncr;

printf("Enter n and r : \n");
scanf("%d%d",&n,&r);

for(i=1;i<=n;++i) factn=factn*i;
for(i=1;i<=r;++i) factr=factr*i;
for(i=1;i<=n-r;++i) factnmr=factnmr*i;
npr=factn/factnmr;
ncr=npr/factr;
printf("Permutation value : %d \n",npr);
printf("Combination Value : %d \n",ncr);
getch();
}
